Use distinct types for fields with distinct roles. This prevents unrelated raw
keys from being substituted into every parameter while keeping all objects
grounded in the observed state.

(define (domain example_typed)
  (:requirements :strips :typing :negative-preconditions)
  (:types actor target status)
  (:predicates (close_to ?who - actor ?what - target) (is_set ?flag - status))
  (:action complete
    :parameters (?who - actor ?what - target ?result - status)
    :precondition (and (close_to ?who ?what) (not (is_set ?result)))
    :effect (is_set ?result)
  )
)

(define (problem example_typed_problem)
  (:domain example_typed)
  (:objects actor_a - actor target_b - target done_c - status)
  (:init (close_to actor_a target_b))
  (:goal (is_set done_c))
)

Observed transitions:
- before `action_a`: `{'worker_a': [[0, 0]], 'station_b': [[1, 0]], 'progress_c': 0, 'finished_d': 0}`
- after `action_a`: `{'worker_a': [[0, 0]], 'station_b': [[1, 0]], 'progress_c': 1, 'finished_d': 0}`
- after subsequent `action_b`: `{'worker_a': [[0, 0]], 'station_b': [[1, 0]], 'progress_c': 1, 'finished_d': 1}`

When an observed quantity is a prerequisite, model the quantity directly and
keep the final status as a separate typed fact. The two operators preserve the
observed ordering instead of collapsing both transitions into one generic
Boolean action.

(define (domain example_staged)
  (:requirements :strips :typing :negative-preconditions :numeric-fluents)
  (:types worker station status)
  (:predicates (at_station ?who - worker ?where - station) (is_set ?flag - status))
  (:functions (progress_c))
  (:action advance
    :parameters (?who - worker ?where - station)
    :precondition (at_station ?who ?where)
    :effect (increase (progress_c) 1)
  )
  (:action finish
    :parameters (?who - worker ?where - station ?result - status)
    :precondition (and (at_station ?who ?where) (>= (progress_c) 1) (not (is_set ?result)))
    :effect (is_set ?result)
  )
)

(define (problem example_staged_problem)
  (:domain example_staged)
  (:objects worker_a - worker station_b - station finished_d - status)
  (:init (at_station worker_a station_b) (= (progress_c) 0))
  (:goal (is_set finished_d))
)

END EXAMPLE 2
