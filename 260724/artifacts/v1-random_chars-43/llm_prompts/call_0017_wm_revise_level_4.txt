from copy import deepcopy

def transition_model(state, action):
    new_state = deepcopy(state)
    new_state["step_count"] = state.get("step_count", 0) + 1

    player = state.get("player", [])
    if not player:
        return new_state

    x, y = player[0]

    directions = {
        "move_left": (-1, 0),
        "move_right": (1, 0),
        "move_up": (0, -1),
        "move_down": (0, 1),
    }

    def positions(key):
        return {tuple(p) for p in new_state.get(key, [])}

    def adjacent_to(key):
        locations = positions(key)
        return any(
            (x + dx, y + dy) in locations
            for dx, dy in directions.values()
        )

    def remove_at(key, pos):
        new_state[key] = [
            p for p in new_state.get(key, [])
            if tuple(p) != pos
        ]

    def add_at(key, pos):
        cells = new_state.setdefault(key, [])
        if list(pos) not in cells:
            cells.append(list(pos))
            cells.sort()

    # Movement: pmzjpl is traversable ground.
    if action in directions:
        dx, dy = directions[action]
        new_state["player_facing"] = [dx, dy]
        destination = (x + dx, y + dy)

        if destination in positions("pmzjpl"):
            new_state["player"] = [[destination[0], destination[1]]]
        return new_state

    facing = state.get("player_facing", [0, 1])
    if len(facing) != 2:
        facing = [0, 1]
    target = (x + facing[0], y + facing[1])

    # Interact with the cell in front of the player.
    if action == "do":

            add_at("pmzjpl", target)

            target in positions("grpoqi")
            and (
                state.get("inv_tpkhxk_wqiqzh", 0) > 0
                or state.get("inv_sgqeje_wqiqzh", 0) > 0
                or state.get("inv_bzracx_wqiqzh", 0) > 0
            )
        ):
            remove_at("grpoqi", target)
            add_at("pmzjpl", target)
            new_state["inv_grpoqi"] = state.get("inv_grpoqi", 0) + 1

        # Iron requires at least a stone-tier pickaxe.
        elif (
            target in positions("bzracx")
            and (
                state.get("inv_sgqeje_wqiqzh", 0) > 0
                or state.get("inv_bzracx_wqiqzh", 0) > 0
            )
        ):
            remove_at("bzracx", target)
            add_at("pmzjpl", target)
            new_state["inv_bzracx"] = state.get("inv_bzracx", 0) + 1

        # The rarest resource requires the highest-tier pickaxe.
        elif (
            target in positions("mwzvua")
            and state.get("inv_bzracx_wqiqzh", 0) > 0
        ):
            remove_at("mwzvua", target)
            add_at("pmzjpl", target)
            new_state["inv_mwzvua"] = state.get("inv_mwzvua", 0) + 1

        return new_state

    # Place an inventory item into an empty ground cell in front.
    if action.startswith("place_"):
        item = action[len("place_"):]
        inv_key = "inv_" + item

        if target not in positions("pmzjpl"):
            return new_state

        costs = {
            "zezroc": ("tpkhxk", 2),  # crafting table
            "ckqpdj": ("sgqeje", 4),  # furnace
        }

        if item in costs:
            material, amount = costs[item]
            material_key = "inv_" + material
            if state.get(material_key, 0) < amount:
                return new_state
            new_state[material_key] = state.get(material_key, 0) - amount
        else:

                return new_state
            new_state[inv_key] = state.get(inv_key, 0) - 1

        remove_at("pmzjpl", target)
        add_at(item, target)
        return new_state

    if action.startswith("make_"):

        parts = product.rsplit("_", 1)
        if len(parts) != 2:
            return new_state

        return new_state

    if action == "sleep":
        new_state["player_sleeping"] = True
        return new_state

    # noop and unsupported/unsuccessful actions only advance step_count.
    return new_state
