(define (domain collect_wood_domain)
  (:requirements :strips :typing :numeric-fluents)
  (:types actor resource status)
  (:constants ach_collect_wood ach_place_table ach_make_wood_pickaxe - status)
  (:predicates
    (adjacent_to ?who - actor ?target - resource)
    (tree_resource ?target - resource)
    (table_ready ?who - actor)
    (achieved ?flag - status)
  )
  (:functions
    (wood_count ?who - actor)
  )
  (:action collect_wood
    :parameters (?who - actor ?target - resource)
    :precondition (and (adjacent_to ?who ?target) (tree_resource ?target))

  )

    :precondition (>= (wood_count ?who) 2)
    :effect (and (decrease (wood_count ?who) 2) (table_ready ?who) (achieved ach_place_table))
  )

    :parameters (?who - actor)
    :precondition (and (table_ready ?who) (>= (wood_count ?who) 1))

  )
)
